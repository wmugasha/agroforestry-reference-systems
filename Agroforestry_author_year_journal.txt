<?xml version="1.0" encoding="utf-8"?>
<style xmlns="http://purl.org/net/xbiblio/csl" version="1.0">
  <info>
    <title>Custom Author (Year) Journal Style</title>
    <id>http://www.example.org/styles/custom-author-year-journal</id>
    <link href="http://www.example.org/styles/custom-author-year-journal" rel="self"/>
    <author>
      <name>Generated by ChatGPT</name>
    </author>
    <category citation-format="numeric"/>
    <updated>2025-12-10T00:00:00+00:00</updated>
  </info>

  <!-- Author-list behavior: list all authors up to 6, then use et al. -->
  <macro name="author-short">
    <names variable="author" and="text">
      <name form="long" and="text" delimiter=", ">
        <name-part name="family"/>
        <name-part name="given" prefix=", "/>
        <label form="short"/>
      </name>
      <et-al term="et-al" prefix=", "/>
    </names>
  </macro>

  <macro name="title">
    <text variable="title"/>
  </macro>

  <macro name="journal">
    <text variable="container-title"/>
    <text variable="volume" prefix=" "/>
    <text variable="page" prefix=":"/>
  </macro>

  <macro name="publisher">
    <group delimiter=", ">
      <text variable="publisher"/>
      <text variable="publisher-place"/>
    </group>
  </macro>

  <macro name="doi-url">
    <group>
      <text variable="DOI" prefix="https://doi.org/"/>
    </group>
  </macro>

  <macro name="accessed">
    <text variable="URL"/>
    <text variable="accessed" prefix=" Accessed " form="text"/>
  </macro>

  <bibliography>
    <layout delimiter="\n">

      <!-- Journal article -->
      <if type="article-journal">
        <text macro="author-short"/>
        <text prefix=" (" variable="issued" form="text" suffix=")"/>
        <text prefix=" " macro="title"/>
        <text prefix=" " macro="journal"/>
        <group delimiter=" ">
          <text macro="doi-url" prefix=" "/>
        </group>
      </if>

      <!-- Article by DOI (no volume/pages) -->
      <if type="article" match="any">
        <text macro="author-short"/>
        <text prefix=" (" variable="issued" form="text" suffix=")"/>
        <text prefix=" " macro="title"/>
        <choose>
          <if variable="container-title">
            <text prefix=" " macro="journal"/>
          </if>
        </choose>
        <group delimiter=" ">
          <text macro="doi-url"/>
        </group>
      </if>

      <!-- Book -->
      <if type="book">
        <text macro="author-short"/>
        <text prefix=" (" variable="issued" form="text" suffix=")"/>
        <text prefix=" " macro="title"/>
        <text prefix=" " macro="publisher"/>
      </if>

      <!-- Book chapter -->
      <if type="chapter">
        <text macro="author-short"/>
        <text prefix=" (" variable="issued" form="text" suffix=")"/>
        <text prefix=" " macro="title"/>
        <text prefix=" In: " variable="editor"/>
        <group delimiter=", ">
          <text variable="container-title"/>
          <text variable="edition" prefix=", "/>
        </group>
        <text prefix=", " macro="publisher"/>
        <text prefix=", pp " variable="page"/>
      </if>

      <!-- Online document -->
      <if type="webpage">
        <text macro="author-short"/>
        <text prefix=" (" variable="issued" form="text" suffix=")"/>
        <text prefix=" " macro="title"/>
        <text prefix=" " variable="source"/>
        <text prefix=" " macro="accessed"/>
      </if>

      <!-- Thesis / Dissertation -->
      <if type="thesis">
        <text macro="author-short"/>
        <text prefix=" (" variable="issued" form="text" suffix=")"/>
        <text prefix=" " macro="title"/>
        <text prefix=" Dissertation, " variable="publisher"/>
      </if>

      <!-- Fallback for other types -->
      <else>
        <text macro="author-short"/>
        <text prefix=" (" variable="issued" form="text" suffix=")"/>
        <text prefix=" " macro="title"/>
      </else>

    </layout>
  </bibliography>

  <!-- Minimal citation (in-text) format: Author (Year) -->
  <citation>
    <layout>
      <text macro="author-short"/>
      <text prefix=" (" variable="issued" form="text" suffix=")"/>
    </layout>
  </citation>

</style>
